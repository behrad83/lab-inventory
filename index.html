ساخت Google Sheet

ساخت API با Apps Script
در Sheet → منوی Extensions → Apps Script → کد زیر را paste کنید → Deploy → Web App → Anyone
function doGet(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sh=ss.getSheets()[0]; var data=sh.getDataRange().getValues(); var keys=data[0]; var rows=data.slice(1).map(function(r){ var obj={};keys.forEach(function(k,i){obj[k]=r[i]}); return obj; }); return ContentService.createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sh=ss.getSheets()[0]; var d=JSON.parse(e.postData.contents); sh.appendRow([d.id,d.code,d.name,d.desc,d.cat,d.qty,d.unit,d.shelf,d.row,d.minQty,d.date]); return ContentService.createTextOutput("ok"); }
